CREATE TABLE error_log (
    log_id         NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    error_code     NUMBER,            -- SQL Error Code
    error_msg      VARCHAR2(4000),    -- Error Message
    log_date       TIMESTAMP DEFAULT CURRENT_TIMESTAMP, -- Time of error
    procedure_name VARCHAR2(100),     -- Name of procedure
    user_name      VARCHAR2(100),     -- User who executed the procedure
    error_data     VARCHAR2(1000)     -- Specific data causing the error (e.g., IDs or relevant values)
);
